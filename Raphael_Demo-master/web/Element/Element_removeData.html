paper = new Raphael(0,0,900,500);
			showthemessage=paper.text(15,400,'Mouse over the Circles.').attr({fill:'#0000ff', 'font-size':24, 'stroke-width':1 ,'text-anchor':'start'});
			function displayData(obj)
			{
				showthemessage.attr( { text: "Circle number : "+obj.data("i")+" has colour "+ obj.data("colour") } );
			}
			var circles=[];
			for(i = 0; i < 5; i++){
			    circles[i]=paper.circle(20 + 50 * i, 20, 20);
			    //Raphael.getColor([value])
				//每次调用返回色谱中的下一个颜色。要重置回红色，调用Raphael.getColor.reset
			    colour=Raphael.getColor(0.75);
			    circles[i].attr({fill: colour});
			    circles[i].data("i", i);
			    circles[i].data("colour", colour);
			    circles[i].mouseover(function (){
			        displayData(this);
			    });
			}

			function removeData(){ 
			    showthemessage.attr( { text: "All colour data was removed.\nNow run the mouse over the circles." } );

				for (i = 0; i < 5; i++){
					/*Element.removeData implemented here*/
			    	circles[i].removeData("colour");
			    }
			} 

			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			/*Element.pause implemented here*/
			paper.widgets.button(paper,500, 400, 75, "Remove\nData", "#800000" , "#FF7F00" ,36,removeData);

			setInfoContent("InfoContent","Example:Element.removeData(key, [value]).","120%");
