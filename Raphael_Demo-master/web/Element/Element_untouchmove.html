paper = new Raphael(0,0,900,500);
			var circle = paper.circle(150,150,60);
			circle.attr({cx:372, cy:150, fill:'#ff0000', r:60, stroke:'#FF00ff', 'stroke-width':5 });

			touchmovehandler=function (e){
				/*Create animate object and include it in the call to the external function, note the call uses the keyword 'this'*/
				animObject=Raphael.animation({ fill: "#FFFF80" }, 1500, 'bounce');
				circle.attr({fill:"#0000ff"});
				circle.animate(animObject);
			};
			circle.touchmove(touchmovehandler);

			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			untouchstart = paper.widgets.button(paper,550, 400, 75, 
				"Stop the\ntouchmove\nnow", "#800000" , "#FF7F00" ,18,
				function (e) { circle.textObject.attr( {text:"Touch move disabled!"} );circle.untouchmove(touchmovehandler); });

			circle.textObject=paper.text(15,400,'Play with touchmove then stop it.')
				.attr({fill:'#0000ff', 'font-size':24, 'stroke-width':1 ,'text-anchor':'start'});
			//setInfoContent("InfoContent","Example:Element.untouchmove(Handler).","120%");
