paper = new Raphael(0,0,900,500);

			//Paper.setStart()
			//创建Paper.set 。
			//调用此方法之后到调用Paper.setFinish方法之前创建的所有元素都会被添加到集合内。
			paper.setStart();
			paper.rect(100,100,100,100,20).attr({fill:"#ff0000"});
			paper.rect(150,150,100,100,20).attr({fill:"#0000FF"});

			//Paper.setFinish()
			//请参照：Paper.setStart 。
			//此方法结束捕获并返回元素集合。
			var roundedrectangles = paper.setFinish();


			roundedrectangles.mousemove(function (e){
				/*Paper.getElementsByPoint implemented here**/
				if (paper.getElementsByPoint.length==2){
					var warn="Congratulations:-You just moused over the rounded rectangles set.";
					alert(warn);
				}
			});

			setContent("InfoContent","Move the mouse over either rectangle
Example:Paper.getElementsByPoint.");
