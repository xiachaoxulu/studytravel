paper = new Raphael(0,0,900,500);
			var img1="../../img/labelraphael.png";
			var img2="../../img/labelsource.png";
			/*Create Graphic Primitives*/
			pic1=paper.image(img1, 150, 100, 50, 50);
			pic2=paper.image(img2, 170, 140, 50, 50);

			pic1.click(function () { pic2.insertBefore (this);});
			pic2.click(function () { pic1.insertBefore (this);});
			paper.text(300, 300,"Click either image to place it on top of the other").attr({fill:'#FF0000', 'font-size':12});
