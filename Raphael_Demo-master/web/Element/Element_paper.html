			//Raphael.el
			//你可以给元素添加自己的方法。
			//当你要修改默认功能或要在一个方法中包装一些常见的变换或属性，这很有用。
			//与画布方法不同，你可以随时重新定义元素的方法。扩展元素的方法，不会影响元素集合。
			Raphael.el.cross = function () {
			    this.attr({fill: "#FF0000"});
			    //这里就用到了Element.papaer
			    //绘制对象的“画布”的内部引用。主要用于插件和元素扩展。
			    this.paper.path("M130,130L170,170M170,130L130,170").attr({stroke: "#FF0000"});
			};
			var ellipse = paper.ellipse(50, 150, 40, 20);

			ellipse.cross();
