function closeAnimate(target)
			{
				target.animate({r:0,"stroke-opacity":0,"fill-opacity":0},3000);
				initialText.attr({text:"Animation.delay: Animation delay and bounce ended.!",fill:"#000",'font-size':24});	
			}
			paper = new Raphael(0,0,900,500);
			animObject=Raphael.animation({ fill: "#FFFF80" }, 1000, 'bounce',function(){ closeAnimate(this);});
			animObject=animObject.delay(2000);
			circle=paper.circle(200, 200, 100);
			circle.animate(animObject);
			initialText=paper.text(15,20,'Animation.delay:Animation delayed 2 Seconds.' ).attr({fill:'#FF0000', 'font-size':20, 'stroke-width':1,'text-anchor':'start' });
			//setInfoContent("InfoContent","Example:Animation.delay.","120%");
