Paper.forEach(callback, thisArg)

为画布上每个元素执行指定函数
如果回调函数返回false则会停止循环。

参数

callback function 函数运行
thisArg object 回调函数的环境
返回：object 画布对象
			paper = new Raphael(0,0,900,500);
			RECT = paper.rect(240, 40, 50, 50, 10).attr({fill:'#008000'});
			PATH = paper.path("M 60,70  L 110,70   A 50,50 0 1,1   60,20   Z").attr({fill:'#C0C000'});
			ELPS = paper.ellipse(250, 150, 40, 20).attr({fill:'#FF0000'});
			var count=0;

			/*Executed for each object*/
			function callback(thisarg){
				if(thisarg.type=="path"){
					var warn="Found a path.(br)Aborting paper.forEach loop.";
					alert(warn);
					//modal.showModalFace(warn,1,[function(){ modal.shutModal(); }],"90%");
					return false;
				}
			}

			//setInfoContent("InfoContent","Example:Paper.forEach.","120%");	
			$('#start').on('click',function(){
				paper.forEach(callback,this );
			})
