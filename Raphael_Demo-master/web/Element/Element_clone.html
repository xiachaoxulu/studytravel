paper = new Raphael(0,0,900,500);
			var r = paper.rect(10, 10, 10, 10);
			var clones=[];
			var z=40;
			while (z--)
			{
				clones[40-z] = r.clone();
				clones[40-z].attr({width:(z+1)*10,y:z*10,fill:Raphael.getColor()});
				paper.text(clones[40-z].attr("width")+20,clones[40-z].attr("y")+5,"#"+(40-z)).attr({"font-size":10});
			}
