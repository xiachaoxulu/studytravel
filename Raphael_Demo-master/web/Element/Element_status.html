Element.status([anim], [value])
获取或设置元素的动画状态。 
			paper = new Raphael(0,0,900,500);
			/*Run an animation from the beginning or start it from a certain point*/
			/***Raphael Widgets***********************************************************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			The function arbitarily called 'onselected' must have key and value parameters.
			The function itself (What ever you called it) is passed into the dropdown/dropdowntransforms constructor.
			*****************************************************************************************************/
			onselected=function(key,value){

				/*****************************************
				Element.status implemented here
				******************************************/
				//Element.status([anim], [value])
				//获取或设置元素的动画状态。
				//第二个参数0 - 1。如果指定，方法就像一个setter，并设置指定动画的状态为指定值。这将导致动画跳转到指定位置。
				circle.animate(animObject);
				circle.status(animObject, key);	
			};
			dropdownSetStatus=paper.widgets.dropdown(paper,  375, 10, 24, 
				["Quarter","Third","Half","Three Quarters"],
			    [0.25,0.33,0.5,0.75],
				onselected,
				{ vmlWidth : 150, colour :"#800000" , backColour:"#FF7F00" , fontSize : 20}) ;

			/*The animation defined*/
			animObject=Raphael.animation({ fill: Raphael.getColor(), r:200 }, 5000, 'bounce');

			/*The element to recieve the animation*/
			circle=paper.circle(200, 200, 100);

			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			resetting=function(){ 
			    circle.remove();
			    circle=paper.circle(200, 200, 100);
			};
			paper.widgets.button(paper,410, 440, 35, "Reset", "#800000" , "#FF7F00" ,24,resetting);
