			var circles=[];
			for(var i=0;i<100;i++){
				cx=Math.floor(Math.random()*800);
				cy=Math.floor(Math.random()*400);
				circles[i]=paper.circle(cx,cy,15);
				/*Randomly assign class*/
				if(cx%2==0){
					/*Element.node implemented here*/
					circles[i].node.setAttribute('class','type1');
				}else{
					/*Element.node implemented here*/
					circles[i].node.setAttribute('class','type2');
				}
			}
			/*************************************************/
			/*Have to define the classes after assigning them*/
			/*************************************************/
			$('.type1').css({ fill:"#00F",stroke:"#F00","stroke-width":5 });
