paper = new Raphael(0,0,900,500);
			RECT = paper.rect(40, 40, 50, 50, 10).attr({fill:'#008000'});
			PATH = paper.path("M 60,70 L  110,70   A 50,50 0 1,1   60,20   Z").attr({fill:'#C0C000'});
			ELPS = paper.ellipse(50, 150, 40, 20).attr({fill:'#FF0000'});

			/***Raphael Widgets***********************************************************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			The function arbitarily called 'onselected' must have key and value parameters.
			The function itself (What ever you called it) is passed into the dropdown/dropdowntransforms constructor.
			*****************************************************************************************************/
			onselected=function(key,value){
				infoval.attr({ text: 'Current transform for '+value +":"});
				applyObjectTranslation(key,transformDesc.attr('text'));
			};
				

			/*Drop down (dropdowntransform) uses the transform string for the key value*/
			dropdowntransform=paper.widgets.dropdown(paper, 210, 10, 24, ["Translate/Rotate","Rotate/Scale","Matrix"], ["t100,100r30,100,100","r30,100,100s2,2","m1.24,0.32,-0.31,1.28,90.67,79.84"],function(key,value){ transformDesc.attr({ text: key });},{ vmlWidth : 150, colour :"#800000" , backColour:"#FF7F00" , fontSize : 20}) ;

			/*Note the 'keys' in drop down 1 are actually Raphael objects*/
			dropdownelement=paper.widgets.dropdown(paper, 410, 10, 24, ["Rectangle","Path","Ellipse"], [RECT,PATH,ELPS],onselected,{ vmlWidth : 150, colour :"#800000" , backColour:"#FF7F00" , fontSize : 20}) ;


			applyObjectTranslation=function(type,transformInformation){
				/*************************************************************************************************************************
				Element.transform is implemented here: The current selected object type (Element) is controlled by the transform command.
				Note that the transformation introduces a new coordinate system, so that attributes like 'x', 'cx' etc remain unchanged.*/
				type.transform(transformInformation);
				setContent("InfoContent",infoval.attr("text")+transformInformation+"(br)(br)Example:Element.transform.");	

			};
			var infoval=paper.text(15,350,'Current transform for Rectangle:').attr({fill:'#0000ff', 'font-size':20, 'stroke-width':1 ,'text-anchor':'start'});
			var transformDesc=paper.text(15,370,'t100,100r30,100,100').attr({fill:'#0000ff', 'font-size':20, 'stroke-width':1 ,'text-anchor':'start'});
			setContent("InfoContent","Example:Element.transform.");
