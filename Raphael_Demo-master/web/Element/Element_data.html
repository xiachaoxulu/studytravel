paper = new Raphael(0,0,900,500);
			showthemessage=paper.text(15,400,'Mouse over the Circles.')
						.attr({fill:'#FF0000', 'font-size':24, 'stroke-width':1 ,'text-anchor':'start'});
			function displayData(obj)
			{
				showthemessage.attr( { text: "Circle number : "+obj.data("i")+" has colour "+ obj.data("colour") } );
			}
			//Create mouse over show data element objects
			for (i = 0; i < 5; i++) 
			{
			    circle=paper.circle(20 + 50 * i, 20, 20);
			    colour=Raphael.getColor(0.75);
			    circle.attr({fill: colour});
			    circle.data("i", i);
			    circle.data("colour", colour);
			    circle.mouseover(function(){
			        displayData(this);
			    });
			}
			//setInfoContent("InfoContent","Example:Element.data(key, [value]).","120%");
