paper = new Raphael(0,0,900,500);
			//Adapted from yet another Dmitry example
			p = paper.path("M100,100C100,80 200,130 200,90"),
			len = p.getTotalLength();


			//ca (The abbreviation for customAttribute) created
			//ca 是 customAttribute 的缩写,下面是定义元素的自定义属性
			paper.ca.distance = function (value) {
			                    value = len / 100 * value;
			                    var point = p.getPointAtLength(value);
			                    return { cx: point.x, cy: point.y };
			                };
			                
			//ca (The abbreviation for customAttribute) implemented                
			var c = paper.circle().attr({distance: 0, r: 5});
			c.animate({distance: 100}, 1000, function () {
				c.animate({distance: 50}, 1000, ">");
			});
			//setInfoContent("InfoContent","Example:Paper.ca.","120%");
