paper = new Raphael(0,0,900,500);
			var stdTextAttributes={fill:'#000000', 'font-size':16, 'stroke-width':1 ,'text-anchor':'start'};

			/*Using the referencable raphael object implemented below with 2 examples*/
			paper.text(50,50,"Rapha\xebl version : "+paper.raphael.version).attr(stdTextAttributes); 
			paper.text(50,70,"Rapha\xebl type : "+paper.raphael.type).attr(stdTextAttributes); 

			var r = paper.circle(240, 160, 160).attr({fill:'#FF0000'}).attr({opacity:0.2});
			var g = paper.circle(400, 160, 160).attr({fill:'#00FF00'}).attr({opacity:0.2});
			
			//将RGB值转换为HSB对象。
			//Raphael.rgb2hsb(r, g, b)
			//返回:HSB对象，格式：{hnumber色相 snumber饱和度 bnumber明度}
			var hsb = paper.raphael.rgb2hsb(r.attr("fill"));
			hsb.h = 1;
			g.attr({fill: paper.raphael.hsb2rgb(hsb).hex});

			paper.text(50,350,'A reserved word which enables access \nto Rapha\xebl\s internal methods.').attr(stdTextAttributes);
