paper = new Raphael(0,0,900,500);

			/*Access the Raphael property "bottom", which contains the reference to all the Primitives*/
			function getobjects(bot){
				var yloc=150;
				var objectList="";
				var counter=1;
				while(bot){
					/*Only want Circles or Elipse*/
					if(bot.attrs.cx!=null)
					{
						objectList=objectList+(counter++)+ ")"+
						"Type : " + bot.type+
						" cx :"+bot.attrs.cx + " cy :"+
						bot.attrs.cy+".\n";			
					}
					bot = bot.next;
				}
				return objectList;
			}
