			/*Our element for transformation*/
			paper = new Raphael(0,0,900,500);
			var tri= paper.path("M 100 100 L 150 100 L 125 140 z").attr({fill: "blue"});
			/*You can define the transform before the animation*/
			tri.transform("t0,40");
			/*Then use a callback to define a further animated transform*/
			tri.animate({ transform:""}, 2000, "elastic",
			function(){
			tri.transform("r60");
			tri.animate({ transform:""}, 2000, "elastic");
			}
			);
