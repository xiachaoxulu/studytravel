paper = new Raphael(0,0,900,500);
			animateNumber=0;

			function closeAnimate(){
				initialText.attr({text:"Animation.repeat #: "+ (++animateNumber)+".",fill:"#000",'font-size':24});	
			}

			animObject=Raphael.animation({ fill: Raphael.getColor(), r:200 },
				 2000, 'bounce',function(){ closeAnimate(this);});
			animObject=animObject.repeat(10);

			circle=paper.circle(200, 200, 100);
			circle.animate(animObject);

			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			/*Stop implemented here*/
			paper.widgets.button(paper,410, 440, 35, "Stop", "#800000" , "#FF7F00" ,24,
				function (e) { circle.stop(); });

			initialText=paper.text(300,20,'Animation repeat has been set forever. Unlesss.....' )
				.attr({fill:'#FF0000', 'font-size':20, 'stroke-width':1,'text-anchor':'start' });
			//setInfoContent("InfoContent","Example:Element.stop.","120%");
