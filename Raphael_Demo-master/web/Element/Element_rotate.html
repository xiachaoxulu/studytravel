			defaultRotate=function (){ 
			    showthemessage.attr( { text: "Default rotation is from the centre.\nIt applies to the objects not the set." } );
			    s.rotate(45);
			};
			paper.widgets.button(paper,105, 275, 100, "Rotate Set\nobjects only\n(1 argument)", "#800000" , "#FF7F00" ,24,defaultRotate);

			locateCentre=function (){ 
			    showthemessage.attr( { text: "Bespoke rotation to locate an alternative centre for rotation." } );
			    s.rotate(45, bbox.x+(bbox.width / 2), bbox.y+(bbox.height / 2));
			};
			paper.widgets.button(paper,315, 275, 100, "Rotate Set\nat centre\n(3 argument)", "#800000" , "#FF7F00" ,24,locateCentre);

			alternativeCentre=function (){

			};

			paper.widgets.button(paper,530, 275, 100, "Rotate Set\nat bottom\nright\n(3 argument)", "#800000" , "#FF7F00" ,24,alternativeCentre);
