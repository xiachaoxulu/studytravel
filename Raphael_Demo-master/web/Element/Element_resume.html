paper = new Raphael(0,0,900,500);
			animateNumber=0;
			function closeAnimate()
			{
				initialText.attr({text:"Animation.repeat #: "+ (++animateNumber)+".",fill:"#000",'font-size':24});	
			}
			animObject=Raphael.animation({ fill: Raphael.getColor(), r:200 }, 2000, 'bounce',function(){ closeAnimate(this);});
			animObject=animObject.repeat(10);
			circle=paper.circle(200, 200, 100);
			circle.animate(animObject);
			/*Pause Animation*/
			//Element.pause([anim])
			//暂停执行动画，稍后可以恢复执行。
			circle.pause();

			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			/*Element.resume implemented here*/
			paper.widgets.button(paper,500, 400, 75, "Resume", "#800000" , "#FF7F00" ,36,function (e) { circle.resume(); });

			initialText=paper.text(300,20,'Animation is paused but will run forever if started.' ).attr({fill:'#FF0000', 'font-size':20, 'stroke-width':1,'text-anchor':'start' });
			setInfoContent("InfoContent","Example:Element.resume.","120%");
