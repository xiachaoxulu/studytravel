paper = new Raphael(0,0,900,500);
			var c = [];
			var z=40;
			while (z--)
			{
			   c[z]=paper.circle((z+1)*10, (z+1)*10, 5).attr({fill:Raphael.getColor()});
			}
			paper.rect(0, 0, 410, 410, 10).attr({fill:"#FFFF00","stroke-width":2,stroke:"#FF0000"});


			var completeModal=function()
			{
				z=40;
				while(z--){
					/*Element.toFront implemented here*/
					c[z].toFront();
				}
			};
			var warn="Warning all new primitives are behind the yellow background.\nClick to put them in front.";
			//modal.showModalFace(warn,1,[function(){ modal.shutModal();completeModal(); },"70%"]);
			//setInfoContent("InfoContent","Example:Element.toFront.","120%");

			$('#front').on('click',function(){
				if(confirm(warn)){
					completeModal();
				}
			})
