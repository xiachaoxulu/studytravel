paper = new Raphael(0,0,900,500);
			paper.rect(200,200,240,80, 10).attr({fill:"#FF0000",opacity:0.4});
			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			setSizeButton480=function(){ 
				paper.widgets.button(paper,230, 90, 70, 
					"Restore the\noriginal\npaper size", "#800000" , "#FF7F00" ,18,
					function(){ paper.setSize(640, 480); });
			}
			setSizeButton240=function(){ 

				paper.setSize(320, 240);
				setSizeButton480();
			}
			paper.widgets.button(paper,90, 90, 70, 
				"Change the\noriginal\npaper size", "#800000" , "#FF7F00" ,18,setSizeButton240);

			//setInfoContent("InfoContent","We are changing the paper size. Play around, but don't forget the reset!(br)(br)Example:Paper.setSize.","120%");
