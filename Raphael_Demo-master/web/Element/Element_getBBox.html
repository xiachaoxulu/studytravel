isWithoutTransformboolean 如果你想得到变换之前的边界，使用true。默认值为 false。 
        	paper = new Raphael(0,0,900,500);
			/*Create a path and put a box around the element*/
			var c = paper.path("M310,250 C310,210 350,210 350,250");
			var bBox = c.getBBox();
			cboundingBox = c.paper.rect(bBox.x,bBox.y,bBox.width,bBox.height).attr({stroke:"#FF0000"});

			/*Clone the path and rotate it from the bottom left and put a box around the element*/
			c2=c.clone();
			c2.rotate(180,bBox.x+bBox.width,bBox.y+bBox.height);
			bBox = c2.getBBox();
			c2boundingBox = paper.rect(bBox.x,bBox.y,bBox.width,bBox.height).attr({stroke:"#0000ff"});
			/*Translate the cloned path and do a new getBBox()*/
			c2.translate(0, 100);
			bBox = c2.getBBox();
			c2boundingBox = paper.rect(bBox.x,bBox.y,bBox.width,bBox.height).attr({stroke:"#0000ff"});
