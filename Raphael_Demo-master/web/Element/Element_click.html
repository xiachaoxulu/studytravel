paper = new Raphael(0,0,900,500);
			var circle = paper.circle(150,150,60);
			circle.attr({cx:372, cy:150, fill:'#0000FF', r:60, stroke:'#FF0000', 'stroke-width':5 });

			function animatedMessage(obj,txt,animObject)
			{
				animObject.message=txt;
				obj.attr({fill:"#ff0000"});
				obj.message=txt;
				obj.animate(animObject);
			};

			circle.click(function(e){
				/*Create animate object and include it in the call to the external function,
				note the call uses the keyword 'this'*/
				//之前学习是Element.animate()和Element.animateWith()
				//这里是Raphael.animation()
				//创建一个动画对象可以传递给Element.animate或Element.animateWith方法。
				animObject=Raphael.animation({ fill: "#FFFF80" }, 1500, 'bounce',
					function(){ 
						paper.text(15,400,this.message)
							.attr({fill:'#FF00FF', 'font-size':30, 'stroke-width':1 ,'text-anchor':'start'});
					});
				animatedMessage(this,"Element.click demo over!",animObject);
			});

			//setInfoContent("InfoContent","Example:Element.click(Handler).","120%");
