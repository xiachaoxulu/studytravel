paper = new Raphael(0,0,900,500);
			paper.rect(200,200,240,80, 10).attr({fill:"#FF0000",opacity:0.4});
			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			setViewBoxButton480 = function(){ 
			paper.widgets.button(paper,230, 90, 70,
				 "Restore the\narea\nCoordinate\nSystem", "#800000" , "#FF7F00" ,18,
				 function(){ paper.setViewBox(0, 0, 640, 480, true); });
			}
			setViewBoxButton240 = function(){ 

				/*
				Paper.setViewBox implemented here
				x number
				new x position, default is 0
				y number
				new y position, default is 0
				w number
				new width of the canvas
				h number
				new height of the canvas
				fit boolean
				true if you want graphics to fit into new boundary box
				*/
				paper.setViewBox(0, 0, 320, 240, true);
				setViewBoxButton480();
			}
			paper.widgets.button(paper,90, 90, 70, 
					"Change the\narea\nCoordinate\nSystem", "#800000" , "#FF7F00" ,18,
					setViewBoxButton240);

			setContent("InfoContent","We are zooming twice as close compared to the initial view with a top left view. Play around with these coordinates though but don't forget the reset!(br)(br)Example:Paper.setViewBox.");
