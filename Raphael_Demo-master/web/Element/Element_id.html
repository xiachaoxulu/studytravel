paper = new Raphael(0,0,900,500);
			RECT = paper.rect(240, 40, 50, 50, 10).attr({fill:'#008000'});
			PATH = paper.path("M 60,70 L  110,70   A 50,50 0 1,1   60,20   Z").attr({fill:'#C0C000'});
			ELPS = paper.ellipse(250, 150, 40, 20).attr({fill:'#FF0000'});

			textfill="";
			bot=paper.bottom;//指向画布上的底层元素
			while (bot){
				/*Element.id implemented here*/
				theID=bot.id;
				unknownElement=paper.getById(theID);
				textfill=textfill+'Unknown element type:'+unknownElement.type+ " Colour:"+unknownElement.attr("fill")+"\n";			
					
				bot = bot.next;
			}
			paper.text(20,250,textfill).attr({fill:'#FF0000', 'font-size':24, 'stroke-width':1 ,'text-anchor':'start'});

			//setInfoContent("InfoContent","Example:Element.id.","120%");
