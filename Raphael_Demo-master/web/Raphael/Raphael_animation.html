paper = new Raphael(0,0,900,500);
			function closeAnimate(target){
				target.animate({r:0,"stroke-opacity":0,"fill-opacity":0},3000);	
			}
			animObject=Raphael.animation({ fill: "#FFFF80" }, 1000, 'bounce',function(){ closeAnimate(this);});
			circle=paper.circle(200, 200, 100);
			circle.animate(animObject);
