paper = new Raphael(0,0,900,500);
			var circle = paper.circle(150,150,60).attr({cx:372, cy:150, fill:'#0000FF', r:60, stroke:'#FF0000', 'stroke-width':5 });

			mousedownhandler=function (e){
				/*Create animate object and include it in the call to the external function,
				note the call uses the keyword 'this'*/
				animObject=Raphael.animation({ fill: "#FFFF80" }, 1500, 'bounce',
					function(){	setContent("InfoContent","Element.mousedown demo over!

Example:Element.unmousedown(Handler).");});
				circle.attr({fill:"#ff0000"});
				circle.animate(animObject);
			};
			circle.mousedown(mousedownhandler);
			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			unmousedown=paper.widgets.button(paper,500, 400, 65, 
				"Stop\nthe\nmousedown\nhandler", "#800000" , "#FF7F00" ,20,
				function (e) { setContent("InfoContent","Mousedown removed!

Example:Element.unmousedown(Handler).");
				circle.unmousedown(mousedownhandler); });

			setContent("InfoContent","Example:Element.unmousedown(Handler).");
