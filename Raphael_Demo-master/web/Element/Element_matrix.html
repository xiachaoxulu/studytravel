			paper = new Raphael(0,0,900,500);
			var exampleHeader="Example:Element.matrix.";

			var PATH = paper.path("M 60,70 L  110,70   A 50,50 0 1,1   60,20   Z").attr({fill:'#C0C000'});

			/***Raphael Widgets***********************************************************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			The function arbitarily called 'onselected' must have key and value parameters.
			The function itself (What ever you called it) is passed into the dropdown/dropdowntransforms constructor.
			*****************************************************************************************************/
			onselected=function(key,value)
			{
				var pathMatrixValues="";
				var raphaelMatrixValues="";
				PATH.transform(key);
				
				//Element.matrix矩阵对象，它代表元素的变换
				/*Element.matrix referenced here*/
				pathMatrixValues='Path Matrix values: '+PATH.matrix.a+" , "+PATH.matrix.b+" , "+PATH.matrix.c+" , "
				+PATH.matrix.d+" , "+PATH.matrix.e+" , "+PATH.matrix.f+"(br)(br)";
				
				//Raphael.matrix(a, b, c, d, e, f)
				//用指定的参数生成矩阵
				M=Raphael.matrix(PATH.matrix.a, PATH.matrix.b, PATH.matrix.c, PATH.matrix.d, PATH.matrix.e, PATH.matrix.f);
				raphaelMatrixValues='M matrix values: '+M.a+" , "+M.b+" , "+M.c+" , "+M.d+" , "+M.e+" , "+M.f+"(br)(br)";
				setInfoContent("InfoContent",pathMatrixValues+raphaelMatrixValues+exampleHeader,"100%","#00f");
			};
			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			***********************************************************/
			dropdowntransforms=paper.widgets.dropdown(paper, 310, 10, 24,  
				["Translate/Rotate","Rotate/Scale","Matrix"], 
				["t100,100r30,100,100","r30,100,100s2,2","m1.24,0.32,-0.31,1.28,90.67,79.84"],
				onselected,
				{ vmlWidth : 150, colour :"#800000" , backColour:"#FF7F00" , fontSize : 20}
			) ;
