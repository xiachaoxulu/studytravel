paper = new Raphael(0,0,900,500);
			var fin = function () {
			    this.attr({ 'opacity': 1 });
			    if(this.flag==null) 
			    {
			    	this.flag = this.paper.rect(120, 10, 100, 100).attr({ 'fill':'#6F6' });
			    }
			    else
			    {
			    	this.flag.show();
			    }
			};
			var fout = function () {
			    this.attr({ 'opacity': .5 });
			    this.flag.hide();
			};
			var target = paper.rect(10, 10, 100, 100).attr({ 'fill': '#66F','opacity': .5 });
			target.hover(fin, fout);
			//setInfoContent("InfoContent","Example:Element.hover.","120%");
