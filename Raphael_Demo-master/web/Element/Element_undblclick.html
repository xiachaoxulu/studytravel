paper = new Raphael(0,0,900,500);
			var colorit = function (){
			    this.attr({ 'fill': Raphael.getColor() });
			};

			var target = paper.rect(10, 10, 100, 100).attr({ 'fill': '#66F','opacity': .5 });
			target.dblclick(colorit);

			/***Raphael Widgets*****************************************
			http://www.irunmywebsite.com/raphael/raphael.widgets.doc.php
			Note: function passed as argument.
			***********************************************************/
			paper.widgets.button(paper,500, 400, 50, 
				"Stop\nDouble\nClick\nHandler", "#800000" , "#FF7F00" ,18,
				function (e) { target.undblclick(colorit); });


			paper.text(15,400,'You can change the colour of the rectangle\nby double clicking on it!')
				.attr({fill:'#0000ff', 'font-size':16, 'stroke-width':1 ,'text-anchor':'start'});
			//setInfoContent("InfoContent","Example:Element.undblclick(handler).","120%");
