paper = new Raphael(0,0,900,500);
			function showAttributes(shape){
			    switch(shape.type){
			        case 'path':
			            shapeInfo=shapeInfo+"\nObject has: "+shape.attrs.path;
			            break;
			        case 'image':
			            shapeInfo=shapeInfo+"\nImage Object with x=" +shape.attrs.x+
			            " y=" +shape.attrs.y+
			            " Width=" +shape.attrs.width+
			            " Height=" +shape.attrs.height+"\n"+
			            " Source=" +shape.attrs.src;
			            break;
			        case 'circle':
			           	shapeInfo=shapeInfo+"\nCircle Object with cx=" +shape.attrs.cx+
			            " cy=" +shape.attrs.cy+
			            " r=" +shape.attrs.r;
			            break;
			        case 'ellipse':
			            shapeInfo=shapeInfo+"\nEllipse Object with x=" +shape.attrs.x+
			            " y=" +shape.attrs.y+
			            " rx=" +shape.attrs.rx+
			            " ry=" +shape.attrs.ry;
			            break;
			        case 'rect':
			        	shapeInfo=shapeInfo+"\nRectangle Object with ";
			            if(shape.attrs.r){ shapeInfo=shapeInfo+"r=" +shape.attrs.r; }
			                
			            shapeInfo=shapeInfo+"x=" +shape.attrs.x+
			            " y=" +shape.attrs.y+
			            " Width=" +shape.attrs.width+
			            " Height=" +shape.attrs.height;
			            break;                     
			    }
			}
			var shapeInfo="Accessing objects implicitly stored...\n";
			/*Existing Objects*/
			paper.image('../../img/labelraphael.png',300, 400, 25, 25);
			paper.image('../../img/labelsource.png',325,400, 25, 25);
			paper.rect(265, 60, 30, 30, 10).attr({fill:'#FF0000'});
			paper.rect(305, 60, 30, 30, 5).attr({fill:'#0000FF'});
			paper.rect(345, 60, 30, 30).attr({fill:'#FF8080'});
			var bot = paper.bottom;
			while (bot){
				showAttributes(bot);
				bot = bot.next;
			}
			paper.rect(5, 150, 620, 210, 10).attr({fill:"#FFFF00"});
			paper.text(10,250,shapeInfo).attr({fill:'#000000', 'font-size':16, 'stroke-width':1, 'text-anchor':'start' }) ;
			//setInfoContent("InfoContent","Example:Paper.bottom.","120%");
